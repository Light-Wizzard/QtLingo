138 {

143 
144  base_node::base_node_ptr root, current;
145  QString text;
146 
147  enum EStates
148  {
149  st_Unstate = 0,
150  st_WaitForStartElement = 0x01,
151  st_WaitForText = 0x02,
152  st_WaitForEndElement = 0x04
153  };
154 
155  int states = st_WaitForStartElement;
156 
157  while(!xmlReader.atEnd())
158  {
159  QXmlStreamReader::TokenType tt = xmlReader.readNext();
160  switch(tt)
161  {
162  case QXmlStreamReader::StartDocument:
163  {
164  root = std::make_shared<document_node>();
165  current = root;
166  } break;
167  case QXmlStreamReader::DTD:
168  {
169  current->add_child(std::make_shared<DTD_node>("<!DOCTYPE TS>"));
170  } break;
171  case QXmlStreamReader::StartElement:
172  {
173  assert(states & st_WaitForStartElement);
174 
175  QString name = xmlReader.name().toString();
176  QXmlStreamAttributes attrs = xmlReader.attributes();
177 
178  if("message" == name)
179  {
180  current = current->add_child(std::make_shared<element_node>(element_node::ent_message, name, attrs));
181  }
182  else if("source" == name)
183  {
184  current = current->add_child(std::make_shared<element_node>(element_node::ent_source, name, attrs));
185  }
186  else if("translation" == name)
187  {
188  current = current->add_child(std::make_shared<element_node>(element_node::ent_translation, name, attrs));
189  }
190  else if("TS" == name)
191  {
192  current = current->add_child(std::make_shared<TS_node>(name, attrs));
193  }
194  else
195  {
196  current = current->add_child(std::make_shared<element_node>(element_node::ent_element, name, attrs));
197  }
198 
199  states = st_WaitForText|st_WaitForStartElement|st_WaitForEndElement;
200  } break;
201  case QXmlStreamReader::Characters:
202  {
203  if(states & st_WaitForText)
204  {
205  text = xmlReader.text().toString();
206  states = st_WaitForEndElement|st_WaitForStartElement;
207  }
208  } break;
209  case QXmlStreamReader::EndElement:
210  {
211  assert(states & st_WaitForEndElement);
212  assert(current->kind() & base_node::nt_Element);
213  //((element_node*)current.get())->set_text(text);
214 
215  reinterpret_cast<element_node*>(current.get())->set_text(text);
216 
217  text.clear();
218  states = st_WaitForStartElement|st_WaitForEndElement;
219  current = current->parent();
220  } break;
221  case QXmlStreamReader::Invalid: { break; }
222  case QXmlStreamReader::EndDocument: { break; }
223  case QXmlStreamReader::Comment: { break; }
224  case QXmlStreamReader::EntityReference: { break; }
225  case QXmlStreamReader::ProcessingInstruction: { break; }
226  case QXmlStreamReader::NoToken: { break; }
227  }
228  }
229 
230  return root;
